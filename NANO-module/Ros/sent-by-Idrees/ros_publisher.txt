#!/usr/bin/env python3

import rospy
from std_msgs.msg import String

str_msg= String()

str_pub= rospy.Publisher("/json_data", String, queue_size=10)
rospy.init_node("JSON_Publisher")


str_msg.data= "{'name':'Sensor1', id: 12}" #basic json example

rate_slow = rospy.Rate(1) #1 Hz

for idx in range(60):
    str_pub.publish(str_msg)
    rate_slow.sleep()
